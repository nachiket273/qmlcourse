\[\begin{split} \sigma^+_i = \left[ \prod_{j< i} (1-2c^\dagger_j c_j) \right] c_i \\ \sigma^-_i = \left[ \prod_{j< i} (1-2c^\dagger_j c_j) \right] c^\dagger_i. \end{split}\]
